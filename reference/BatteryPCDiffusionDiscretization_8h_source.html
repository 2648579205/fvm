19 template<class T>
20 inline T harmonicAverageVector(const T& x0, const T& x1)
21 {
22  const T sum = x0+x1;
23  T harmAvg = sum;
24  int vectorLength = 0;
25  NumTypeTraits<T>::getShape(&vectorLength);
26  for (int v=0; v<vectorLength; v++)
27  {
28  if (sum[v] != 0.0)
29  harmAvg[v] = 2.0*x0[v]*x1[v]/sum[v];
30  }
31  return harmAvg;
32 }
