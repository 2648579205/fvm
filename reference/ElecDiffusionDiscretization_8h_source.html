115  if (((ibType[c0] == Mesh::IBTYPE_FLUID)

129  }
130  }

141  const T_Scalar diffCoeff = faceDiffusivity*diffMetric;
142  const VectorT3 secondaryCoeff = faceDiffusivity*(faceArea[f]-ds*diffMetric);
143
