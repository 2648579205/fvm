19 template<class T>
20 inline T harmonicAverage(const T& x0, const T& x1)
21 {
22  const T sum = x0+x1;
23  if (x0+x1 != NumTypeTraits<T>::getZero())
24  return 2.0*x0*x1/sum;
25  else
26  return sum;
27 }
